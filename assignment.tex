\begin{enumerate}

\item 
\textbf{Integration}:
\[
\int_{0}^{1} x^2 e^x \, dx
\]

\item A coin is tossed twice. The following table shows the probability distribution of the number of tails:

\[
\begin{array}{|c|c|c|c|}
\hline
X & 0 & 1 & 2 \\
\hline
P(X) & K & 6K & 9K \\
\hline
\end{array}
\]

(a) Find the value of $K$.

(b) Determine whether the coin toss is biased or unbiased, and provide a justification for your answer.

\item 
Find the general solution for this differential equation:
\[
\sec^2 x \tan y \, dx + \sec^2 y \tan x \, dy = 0
\]

\item 
If the area of the region bounded by the line $y = mx$ and the curve $x^2 = y$ is $\frac{32}{3}$ sq. units, then find the positive value of $m$ using integration.

\item 
Given vectors:
\[
\vec{a} = 2\hat{i} - \hat{j} + \hat{k}, \quad \vec{b} = \hat{i} + \hat{j} - 2\hat{k}, \quad \vec{c} = \hat{i} + 3\hat{j} - \hat{k}
\]
and the projection of the vector $\vec{c} + \lambda \vec{b}$ on vector $\vec{a}$ is $2\sqrt{6}$, find the value of $\lambda$.

\item 
(a) \textbf{Find}:
\[
\int \frac{1}{e^x+1} \, dx
\]

\textbf{OR}

(b) \textbf{Evaluate}:
\[
\int_{1}^{4} \left| |x| + |3-x| \right| \, dx
\]

\item 
\textbf{Evaluate}:
\[
\int_{-3}^{3} \frac{x^4}{1+e^x} \, dx
\]

\item 
(a) Find the particular solution of the differential equation:
\[
x\frac{dy}{dx} + y + \frac{1}{1+x^2} = 0
\]
given that $y(1) = 0$.

\textbf{OR}

(b) Find the general solution of the differential equation:
\[
x(y^3+x^3) \, dy = (2y^4+5x^3y) \, dx
\]

\item 
Find:
\[
\int \frac{dx}{\sqrt{4x-x^2}}
\]

\item 
Find the general solution of the following equation:
\[
\frac{dy}{dx} = e^x - yx^2e^{-y}
\]

\item 
Let $X$ be a random variable which assumes values $x_1, x_2, x_3, x_4$ such that
\[
2P(X=x_1) = 3P(X=x_2) = P(X=x_3) = 5P(X=x_4)
\]
Find the probability distribution of $X$.

\item 
(a) \textbf{Find}:
\[
\int e^x \sin(2x) \, dx
\]

\textbf{OR}

(b) Find:
\[
\int \frac{2x}{(x^2+1)(x^2+2)} \, dx
\]

\item 
\textbf{Evaluate}:
\[
\int_{1}^{3} \frac{\sqrt{x}}{\sqrt{x}+\sqrt{4-x}} \, dx
\]

\item 
(a) Solve the following differential equations:
\[
(y-\sin^2 x) \, dx + \tan(x) \, dy = 0
\]

(b) Find the general solution of the differential equation:
\[
(x^3+y^3) \, dy = x^2y \, dx
\]

\item 
(a) \textbf{Find}:
\[
\int \frac{1}{\sqrt{12+4x-x^2}} \, dx
\]

\textbf{OR}

(b) Find:
\[
\int \frac{xe^x}{(x+4)^5} \, dx
\]

\item 
Find the general solution of the following differential equation:
\[
(4+y^2)(3+\log x) \, dx + x \, dy = 0
\]

\item 
\textbf{Evaluate}:
\[
\int_{0}^{\frac{\pi}{3}} \left| \cos(3x) \right| \, dx
\]

\item 
(a) Find the general solution of the following differential equation:
\[
2x e^{\frac{y}{x}} \, dy + (x-2y e^{\frac{y}{x}}) \, dx = 0
\]

\textbf{OR}

(b) Find the particular solution of the differential equation:
\[
(2x^2+y) \frac{dx}{dy} = x
\]
given that $y=2$ when $x=1$.

\item 
Find:
\[
\int \frac{x^2+x+1}{(x+1)(x^2+4)} \, dx
\]

\item 
(a) Find the area bounded by the ellipse $x^2+4y^2=16$ and the ordinates $x=0$ and $x=2$, using integration.

\textbf{OR}

(b) Find the area of the region $\{(x,y): x^2 \leq y \leq x\}$, using integration.

\item 
\[
\int_{0}^{\frac{\pi}{2}} \frac{1}{1+\sqrt{\cot x}} \, dx
\] 
is equal to:
\begin{enumerate}
\item $\frac{\pi}{3}$
\item $\frac{\pi}{6}$
\item $\frac{\pi}{4}$
\item $\frac{\pi}{2}$
\end{enumerate}

\item 
Find:
\[
\int \frac{(x+2)(x+2\log x)^3}{x} \, dx
\]

\item 
\[
\int_{0}^{\frac{\pi}{2}} \log(\tan x) \, dx
\]

\textbf{OR}

\[
\int_{-1}^{2} |x| \, dx
\]

\item 
Find:

$\int\limits x^2 log x . dx$

\item 
Find the general solution of the following differential equation :\\

$\frac{dy}{dx}= (1+x)(1+y)$
\begin{center}
    \textbf{OR}
\end{center}
 Find the integrating factor for the following differential equation :\\

 $\frac{dy}{dx}+ y \cot x = 2x+x^2 \cot x (x \neq 0)$
 
 \item 
 Find:\\

 $\int\limits\frac{x}{(x-1)^2(x+2)} dx$

 \item 
 Find the following differential equation :\\

 $x\cos(\frac{y}{x})\frac{dy}{dx}=y \cos (\frac{y}{x})+x $

\end{enumerate}
